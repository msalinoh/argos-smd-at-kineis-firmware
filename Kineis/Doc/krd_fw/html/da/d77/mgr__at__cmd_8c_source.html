67 /**
68  * @brief API used to extract the latest AT cmds from the incoming received data stream.
69  *
70  * This fct is looking for the start pattern ("AT+") and the end pattern ("\r\n") in the stream.
71  * If several frames are received in a row, only the last AT cmd is extracted
72  *
73  * Once AT cmd is found, it is removed from the stream and stored into the FIFO if not full.
74  *
75  * As this fct is the entry point into FW, it is real-time critical and may be more robust. Some
76  * design limitations are describerd below.
77  *
78  * @attention This fct may be called from ISR context
79  *
80  * @note Any garbage after the found AT cmd is also removed from stream. But the garbage before
81  * remains in the original stream.
82  *
83  * @param[in,out] pu8_RxBuffer pointer to start of RX buffer
84  * @param[in,out] pi16_nbRxValidChar number of valid charecters in RX buffer
85  *
86  * @retval true if a valid frame was extracted, false if nothing was extracted

97  /* Process buffer only once termination caracter has been found */
98  if (C != '\n')
99  return false;
100 
101  /* AT cmd starts with 'AT+' and ends with '\r\n'. Minimum packet length is ("AT+\r\n") */
102  if (*pi16_nbRxValidChar < 5)
103  return false;
104 
105  /* Search beginning of AT cmd from the end of the buffer */
106  for (idxStart = *pi16_nbRxValidChar - 1; idxStart >= 0; idxStart--)
107  if (pu8_RxBuffer[idxStart] == 'A' &&
108  pu8_RxBuffer[idxStart + 1] == 'T' &&
109  pu8_RxBuffer[idxStart + 2] == '+') {
110  isFirstCharDetected = true;
111  break;
112  }
113 
114  /* Return if we can not find the beginning of the frame*/
115  if (!isFirstCharDetected)
116  return false;
117 
118  /* Find last characters of the frame : '\r\n' */
119  for (idxEnd = idxStart + 2; idxEnd <= *pi16_nbRxValidChar; idxEnd++) {
120  if (pu8_RxBuffer[idxEnd - 2] == '\r' && pu8_RxBuffer[idxEnd - 1] == '\n') {
121  isEOLdetected = true;
122  break;
123  }
124  }
125  /* Return if we can not find the EOL characters */
126  if (!isEOLdetected)
127  return false;
128 
129  /* As some AT cmd was consummed from the end at idxStart position,
130  * idxStart characters remaing to be parsed at the beginning of the buffer.
131  * Assume they are all valid.
132  *
133  */
134  *pi16_nbRxValidChar = idxStart;
135 
136  /* In case FIFO is full (read index is reached), skip this new AT CMD. We need to wait for
137  * FW to consume the previous AT CMDs before.
138  */
139  if ( ((‪s_atcmdfifo.‪u8_widx+1) % ‪FIFO_MAX_SIZE) == (‪s_atcmdfifo.‪u8_ridx % ‪FIFO_MAX_SIZE))
140  return true;
141  /* Set the frame in the UART fifo */
142  i16_atcmdLen = idxEnd - idxStart;

144  * detected (reserve last character for end-of-string'\0').
145  */
146  if ((i16_atcmdLen + 1) > ‪FRAME_MAX_LEN)
147  i16_atcmdLen = ‪FRAME_MAX_LEN - 1;
148  memcpy(‪s_atcmdfifo.‪au8_fifo[‪s_atcmdfifo.‪u8_widx % ‪FIFO_MAX_SIZE],
149  &pu8_RxBuffer[idxStart], i16_atcmdLen);
150  ‪s_atcmdfifo.‪au8_fifo[‪s_atcmdfifo.‪u8_widx % ‪FIFO_MAX_SIZE][i16_atcmdLen] = '\0';
151  /* Increment write index to next free position */
152  ‪s_atcmdfifo.‪u8_widx++;
153 
154  /* In case read index is reached, skip oldest stored AT cmd to let a newest one to come */
155  if (‪s_atcmdfifo.‪u8_widx % ‪FIFO_MAX_SIZE == ‪s_atcmdfifo.‪u8_ridx % ‪FIFO_MAX_SIZE)
156  ‪kns_assert(0); // should no more happen

158

160 }

162 /**

164  *
165  * AT cmds with "=?\r\n" (e.g. "AT+ID=?\r\n") is a status-type command. It is used to get some
166  * information from the device.

169  * the device (e.g. "AT+TX=<...>\r\n").
170  *
171  * @note This fct may be called from ISR context

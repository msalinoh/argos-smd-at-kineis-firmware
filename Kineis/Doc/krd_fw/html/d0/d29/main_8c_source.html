133  * @brief Fill-up heap and stack RAM space to track overflows (with 0x55555555 and 0xAAAAAAAA)
134  */
135 void mspFillup(void)
136 {
137  extern uint8_t _end; /* Symbol defined in the linker script */
138  extern uint8_t _estack; /* Symbol defined in the linker script */
139  extern uint32_t _Min_Stack_Size; /* Symbol defined in the linker script */
140  uint32_t *stack_lower_limit = (uint32_t*)((uint32_t)&_estack - (uint32_t)&_Min_Stack_Size);
141  uint32_t *stack_upper_limit = (uint32_t*)__get_MSP();
142  uint32_t *heap_lower_limit = (uint32_t*)((uint32_t)&_end);
143 
144  /* Paint heap */
145  for (
146  heap_lower_limit = (uint32_t*)((uint32_t)&_end);
147  heap_lower_limit < stack_lower_limit;
148  heap_lower_limit++)
149  {
150  *(heap_lower_limit) = 0x55555555;
151  }
152  /* Paint stack */
153  for (
154  stack_lower_limit = (uint32_t*)((uint32_t)&_estack - (uint32_t)&_Min_Stack_Size);
155  stack_lower_limit < stack_upper_limit;
156  stack_lower_limit++)
157  {
158  *(stack_lower_limit) = 0xAAAAAAAA;
159  }
160 }

140  * The standalone application should be much more simple with a real OS
141  */
142  static
143  ‪__attribute__((__section__(".retentionRamData")))
144  uint8_t state;
145 
146  /** Static variable used with rand_r() function to generate random USER DATA in messages
147  * sent to Kineis stack.
148  * This will work also with STANDBY LPM as the seed tis mapped in SRAM2 memory here
149  * @note using rand() instead of rand_r() may lead to generate several messages with the
150  * same USER DATA payload
151  */
152  static
153  ‪__attribute__((__section__(".retentionRamData")))
154  uint32_t seed;
155 
156  uint16_t ‪idx;
157  struct ‪KNS_MAC_appEvt_t appEvt;
158  uint8_t buffer_tx[‪KNS_MAC_USRDATA_MAXLEN];
159 
160  switch (state) {

162  struct ‪KNS_CFG_radio_t device_radio_cfg;
163 
164  /** Initialize buffer with random data */
165  for (‪idx = 0; ‪idx < sizeof(buffer_tx); ‪idx++)
166  buffer_tx[‪idx] = rand_r(((unsigned int *)&seed));
167 
168  /** ---- OPTIONAL BLOCK ---- START -- needed if radio conf not hardcoded ----- */
169  ‪kns_assert(‪KNS_CFG_getRadioInfo(&device_radio_cfg) == ‪KNS_STATUS_OK);
170 
171  appEvt.‪id = ‪KNS_MAC_SEND_DATA;

173  appEvt.‪data_ctxt.‪usrdata[‪idx] = buffer_tx[‪idx];
174  appEvt.‪data_ctxt.‪sf = ‪KNS_SF_NO_SERVICE;
175 
176  switch (device_radio_cfg.‪modulation) {

179  appEvt.‪data_ctxt.‪usrdata_bitlen = 192;
180  break;
181  case (‪KNS_TX_MOD_LDA2L):
182  appEvt.‪data_ctxt.‪usrdata_bitlen = 196;
183  break;
184  case (‪KNS_TX_MOD_VLDA4):

187  case (‪KNS_TX_MOD_LDK):

189  break;
190  default:
191  ‪kns_assert(0); // wrong modulation flashed into device
192  break;
193  }
194  /** ---- OPTIONAL BLOCK ---- END --------------------------------------------- */
195 
196  ‪MGR_LOG_DEBUG("[%s] request to send 0x", __func__);
197  ‪MGR_LOG_array(appEvt.‪data_ctxt.‪usrdata, (appEvt.‪data_ctxt.‪usrdata_bitlen+7)>>3);
198 
199  ‪TEST_ASSERT(‪KNS_Q_push(‪KNS_Q_DL_APP2MAC, (void *)&appEvt) == ‪KNS_STATUS_OK);
200 
201  state++; /* go to next state, and return to let higher task to process event */
202  return;
203  }
204  case 1: { /** Wait for Kineis stack replying:
205  * * OK to send frame
206  * * TX done for previous transmit
207  */
208  enum ‪KNS_status_t status = ‪KNS_STATUS_OK;
209  struct ‪KNS_MAC_srvcEvt_t srvcEvt;
210 
211  for (status = ‪KNS_Q_pop(‪KNS_Q_UL_MAC2APP, (void *)&srvcEvt);
212  status != ‪KNS_STATUS_QEMPTY;
213  status = ‪KNS_Q_pop(‪KNS_Q_UL_MAC2APP, (void *)&srvcEvt)) {
214  if (status == ‪KNS_STATUS_OK) {
215  switch (srvcEvt.id) {
216  case (‪KNS_MAC_OK):
217  if (srvcEvt.app_evt == ‪KNS_MAC_SEND_DATA) {
218  ‪MGR_LOG_DEBUG("[%s] OK to send 0x", __func__);
219  ‪MGR_LOG_array(srvcEvt.tx_ctxt.data,
220 // (srvcEvt.tx_ctxt.data_bitlen+7)>>3);
221  4); // limit to 4 bytes for real-time
222  }
223  break;
224  case (‪KNS_MAC_TX_DONE):
225  ‪MGR_LOG_DEBUG("[%s] TX done for 0x", __func__);
226  ‪MGR_LOG_array(srvcEvt.tx_ctxt.data,
227  (srvcEvt.tx_ctxt.data_bitlen+7)>>3);
228  ‪TEST_PASS();
229  state++;
230  break;
231  case (‪KNS_MAC_TX_TIMEOUT):
232  ‪MGR_LOG_DEBUG("[%s] TX timeout for 0x", __func__);
233  ‪MGR_LOG_array(srvcEvt.tx_ctxt.data,
234  (srvcEvt.tx_ctxt.data_bitlen+7)>>3);
235  ‪TEST_FAIL();
236  state++;
237  break;
238  case (‪KNS_MAC_ERROR):
239  /** MAC ERROR can occure when FIFO is full, as expected per
240  * this application
241  */
242  if (srvcEvt.app_evt == ‪KNS_MAC_SEND_DATA) {
243  ‪MGR_LOG_DEBUG("[%s] cannot send 0x", __func__);
244  ‪MGR_LOG_array(srvcEvt.tx_ctxt.data,
245 // (srvcEvt.tx_ctxt.data_bitlen+7)>>3);
246  4); // limit to 4 bytes for real-time
247  }
248  ‪TEST_FAIL();
249  state++;
250  break;
251  default:
252  ‪TEST_FAIL();
253  break;
254  }
255  }
256  }
257  return;
258  }
259  case 2: { /** Wait for Kineis stack replying:*/
260  break;
261  }
262  default:
263  ‪kns_assert(0);
264  break;
265  }
